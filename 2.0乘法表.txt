public static void main(String[] args) {
		for(int a=1;a<10;a++){//行
			for(int b=1;b<=a;b++){//列
				System.out.print(a+"*"+b+"= "+a*b+"  ");//输出
			}
			System.out.println();//换行
		}
	}
